
    // Parameterized constructor
    public Agent(String name, int age, String codename, boolean isUndercover, String assignment, String specialty) {
        this.name = name;
        this.age = age;
        this.codename = codename;
        this.isUndercover = isUndercover;
        this.assignment = assignment;
        this.specialty = specialty;
    }

    

    // Method to update assignment
    void updateAssignment(String newAssignment) {
        this.assignment = newAssignment;
    }
}

        Agent specialOpsAgent = new Agent(name, age, codename, isUndercover, assignment, specialty);

        // Introduce the specialOpsAgent
        specialOpsAgent.introduce();

        
        // Re-introduce the specialOpsAgent with the updated assignment
        specialOpsAgent.introduce();

    }
